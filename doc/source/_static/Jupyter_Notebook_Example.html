In [5]:
lens = PlanoConvexLens(centre=(0,0,20),
                      direction=(0,0,1),
                      diameter=25,
                      curvature=100,
                      n_inside=1.6)

src = ConfocalRayFieldSource(centre=(0,0,0),
                            direction=(0,0,1),
                            angle=5.0,
                            angle_step=0.5,
                            E_vector=(1,0,0),
                            working_dict=100.0,
                            display='wires')

probe = EFieldPlane(source=src,
                   width=0.05,
                   height=1.0,
                   size=100,
                   centre=(0,0,77.4),
                   direction=(0,1,0))

model = RayTraceModel(optics=[lens],
                     sources=[src],
                     probes=[probe])
